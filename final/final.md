function html2urls(html) {
    var r = /\shref\s*=\s*['"](.*?)['"]/g 
    // \s單一或多個空白字元 href 即href \s* 多個或單一個空白 ['"]'"的聯集 
    // .*? 比對多個任意字元且可為0個或1個 ()在這裡不確定意思 不像是 Capturing Parentheses捕捉.*? 以方便之後使用 推測是括弧起來增加可讀性 g即全域比對
    var urls = []
    while (true) {
        let m = r.exec(html)
        if (m == null) break
        urls.push(m[1])
    }
    return urls
}
